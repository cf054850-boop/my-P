פונקציות ב-JavaScript – מבוא

פונקציה היא בלוק קוד שניתן לקרוא לו מספר פעמים, בדיוק כמו ב-C#.

מבנה בסיסי של פונקציה:

// הצהרה על פונקציה function sayHello() { console.log("שלום עולם!"); } // קריאה לפונקציה sayHello(); // "שלום עולם!" sayHello(); // "שלום עולם!"

רכיבי פונקציה:

  • function - מילת מפתח להצהרה
  • sayHello - שם הפונקציה
  • () - סוגריים לפרמטרים
  • {} - גוף הפונקציה
ב-C# כותבים: void SayHello() { }, ב-JavaScript: function sayHello() { } - ללא הצהרת טיפוס

פונקציות שמחזירות ערך ומקבלות ארגומנטים

פונקציות יכולות לקבל ארגומנטים (פרמטרים) ולהחזיר ערכים:

// פונקציה עם פרמטר אחד function greet(name) { console.log("שלום " + name); } greet("יצחק"); // "שלום יצחק" greet("שרה"); // "שלום שרה" // פונקציה שמחזירה ערך function add(a, b) { return a + b; } let result = add(5, 3); console.log(result); // 8 // פונקציה מורכבת יותר function calculateArea(width, height) { let area = width * height; return area; } console.log(calculateArea(10, 5)); // 50

נקודות חשובות:

  • אין צורך להצהיר על טיפוס הפרמטרים
  • אין צורך להצהיר על טיפוס ההחזרה
  • return מסיים את ביצוע הפונקציה
  • פונקציה ללא return מחזירה undefined
ב-C# כותבים: int Add(int a, int b), ב-JavaScript: function add(a, b) - ללא טיפוסים

ארגומנטים ברירת מחדל

ב-JavaScript ניתן להגדיר ערכי ברירת מחדל לפרמטרים:

// ללא ברירת מחדל - בעיה function greet1(name) { console.log("שלום " + name); } greet1(); // "שלום undefined" - לא טוב! // עם ברירת מחדל - פתרון function greet2(name = "אורח") { console.log("שלום " + name); } greet2(); // "שלום אורח" greet2("יוסף"); // "שלום יוסף" // דוגמה עם מספרים function multiply(a, b = 1) { return a * b; } console.log(multiply(5, 3)); // 15 console.log(multiply(5)); // 5 (b=1) // מספר פרמטרים עם ברירת מחדל function createUser(name = "אנונימי", age = 18, country = "ישראל") { return `${name}, בן ${age}, מ${country}`; } console.log(createUser()); // "אנונימי, בן 18, מישראל" console.log(createUser("יהושע")); // "יהושע, בן 18, מישראל" console.log(createUser("יהושע", 25, "ארה״ב")); // "יהושע, בן 25, מארה״ב"
גם ב-C# וגם ב-JavaScript יש תמיכה בפרמטרים עם ברירת מחדל, התחביר דומה

פונקציה בתוך משתנה

ב-JavaScript, פונקציות הן אובייקטים מדרגה ראשונה - ניתן לשמור אותן במשתנים:

// Function Declaration (הצהרה רגילה) function add1(a, b) { return a + b; } // Function Expression (ביטוי פונקציה) const add2 = function(a, b) { return a + b; }; // שתיהן עובדות אותו דבר console.log(add1(3, 4)); // 7 console.log(add2(3, 4)); // 7 // פונקציה אנונימית (ללא שם) const greet = function(name) { return "שלום " + name; }; console.log(greet("משה")); // "שלום משה"

ההבדלים:

  • function add1() - ניתנת ל-hoisting (ראה שקף הבא)
  • const add2 = function() - לא ניתנת ל-hoisting
  • שתיהן תקפות ושימושיות
ב-C# אין אפשרות פשוטה לשמור פונקציה במשתנה (יש delegates ו-lambdas), ב-JavaScript זה טבעי

Function Hoisting

Hoisting פירושו "הרמה" - JavaScript "מרים" הצהרות פונקציות לתחילת הקוד.

// זה עובד! למרות שהפונקציה מוגדרת אחרי הקריאה sayHello(); function sayHello() { console.log("שלום!"); } // JavaScript "רואה" את זה כך: // function sayHello() { ... } ← הועבר לראש // sayHello(); // אבל זה לא עובד עם Function Expression: // greet(); // ✗ שגיאה! Cannot access 'greet' before initialization const greet = function() { console.log("שלום"); };

למה זה קורה?

  • JavaScript סורקת את כל הקוד לפני הרצה
  • היא "מרימה" הצהרות פונקציות לתחילה
  • זה מאפשר לקרוא לפונקציה לפני ההגדרה שלה
  • Function Expressions לא עוברות hoisting כי הן משתנים
ב-C# חייבים להגדיר פונקציה לפני השימוש בה, ב-JavaScript function declarations עוברות hoisting

Arrow Functions (פונקציות חץ)

Arrow Functions הן תחביר קצר וקריא יותר לכתיבת פונקציות:

// תחביר רגיל const add1 = function(a, b) { return a + b; }; // תחביר Arrow Function const add2 = (a, b) => { return a + b; }; // תחביר מקוצר (return משתמע) const add3 = (a, b) => a + b; // כולן עושות אותו דבר console.log(add1(3, 4)); // 7 console.log(add2(3, 4)); // 7 console.log(add3(3, 4)); // 7 // עוד דוגמאות const square = x => x * x; // פרמטר אחד - ללא סוגריים const sayHello_ = () => console.log("hello"); // ללא פרמטרים console.log(square(5)); // 25 sayHello_(); // "hello"

מתי להשתמש ב-Arrow Functions?

  • פונקציות קצרות וקריאות
  • callbacks (מועברות לפונקציות אחרות)
  • כאשר רוצים קוד מינימליסטי
  • יש הבדלים ב-this (נלמד בהמשך)
Arrow Functions ב-JavaScript דומות ל-Lambda Expressions ב-C#: (a, b) => a + b

Strings ב-JavaScript

JavaScript מספקת מתודות רבות לעבודה עם מחרוזות:

let text = "JavaScript is Amazing"; // אורך המחרוזת console.log(text.length); // 21 // גישה לתו לפי אינדקס console.log(text[0]); // "J" console.log(text.charAt(0)); // "J" // המרה לאותיות גדולות/קטנות console.log(text.toUpperCase()); // "JAVASCRIPT IS AMAZING" console.log(text.toLowerCase()); // "javascript is amazing" // חיפוש במחרוזת console.log(text.indexOf("is")); // 11 console.log(text.includes("Amazing")); // true console.log(text.startsWith("Java")); // true console.log(text.endsWith("ing")); // true // חיתוך מחרוזת console.log(text.slice(0, 10)); // "JavaScript" console.log(text.substring(0, 10)); // "JavaScript" // החלפת טקסט console.log(text.replace("Amazing", "Cool")); // "JavaScript is Cool" // פיצול לפי מפריד let words = text.split(" "); console.log(words); // ["JavaScript", "is", "Amazing"] // הסרת רווחים מהקצוות let str = " hello "; console.log(str.trim()); // "hello"
מתודות String ב-JavaScript דומות ל-C# (ToUpper, Contains, Replace וכו'), אבל בשמות camelCase

Date Object

האובייקט Date מאפשר עבודה עם תאריכים ושעות:

// יצירת תאריך - עכשיו let now = new Date(); console.log(now); // לדוגמה: 2026-02-01T10:30:00.000Z // יצירת תאריך ספציפי let birthday = new Date(2000, 0, 15); // 15 בינואר 2000 (חודש מתחיל מ-0!) let newYear = new Date("2026-01-01"); // שליפת נתונים מתאריך console.log(now.getFullYear()); // 2026 console.log(now.getMonth()); // 1 (פברואר - מתחיל מ-0!) console.log(now.getDate()); // 1 (יום בחודש) console.log(now.getDay()); // 6 (יום בשבוע: 0=ראשון, 6=שבת) console.log(now.getHours()); // שעה (0-23) console.log(now.getMinutes()); // דקות (0-59) console.log(now.getSeconds()); // שניות (0-59) // המרה למחרוזת console.log(now.toDateString()); // "Sat Feb 01 2026" console.log(now.toLocaleDateString()); // "2/1/2026" (תלוי בלוקל) console.log(now.toLocaleTimeString()); // "10:30:00 AM"

שימושים נפוצים:

// חישוב הפרש בין תאריכים let date1 = new Date("2026-01-01"); let date2 = new Date("2026-02-01"); let diff = date2 - date1; // הפרש במילישניות let days = diff / (1000 * 60 * 60 * 24); console.log(days + " ימים"); // "31 ימים" // הצגת תאריך מעוצב let today = new Date(); let formatted = `${today.getDate()}/${today.getMonth()+1}/${today.getFullYear()}`; console.log(formatted); // "1/2/2026"
Date ב-JavaScript = DateTime ב-C#, אבל ב-JS חודשים מתחילים מ-0 (0=ינואר, 11=דצמבר)
